<%*
// 获取当前日期
const moment = window.moment;
const today = moment();

// 判断日期属性
const isWeekend = today.day() === 0 || today.day() === 6; // 周日=0, 周六=6
const isMonthEnd = today.date() === today.daysInMonth();
const isQuarterEnd = today.month() % 3 === 2 && today.date() === today.daysInMonth();
const isYearEnd = today.month() === 11 && today.date() === 31;

// 确定目标文件夹
let targetFolder = "日记/每日"; // 默认每日
let templateFile = "";

if (isYearEnd) {
    targetFolder = "日记/年度";
    templateFile = "模板/日记模板/Yearly Notes";
} else if (isQuarterEnd) {
    targetFolder = "日记/季度";
    templateFile = "模板/日记模板/Quarterly Notes";
} else if (isMonthEnd) {
    targetFolder = "日记/每月";
    templateFile = "模板/日记模板/Monthly Notes";
} else if (isWeekend) {
    targetFolder = "日记/每周";
    templateFile = "模板/日记模板/Weekly Notes";
} else {
    targetFolder = "日记/每日";
    templateFile = "模板/日记模板/Daily Note";
}

// 生成文件名和路径
const fileName = today.format("YYYY-MM-DD");
const filePath = `${targetFolder}/${fileName}.md`;

// 检查文件是否已存在，避免覆盖
const existingFile = app.vault.getAbstractFileByPath(filePath);
if (existingFile) {
    new Notice(`文件 ${filePath} 已存在`);
    return;
}

// 读取模板内容
const templateContent = await tp.file.include(templateFile);

// 创建文件并写入内容
await app.vault.create(filePath, templateContent);

// 打开新创建的文件
const newFile = app.vault.getAbstractFileByPath(filePath);
await app.workspace.getLeaf().openFile(newFile);
%>
